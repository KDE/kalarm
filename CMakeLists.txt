    include(ECMAddAppIcon)
    include(ECMQtDeclareLoggingCategory)

    set(MAILCOMMON_LIB_VERSION_LIB "5.2.40")
    set(PIMCOMMON_LIB_VERSION_LIB "5.2.40")
    set(LIBKDEPIM_LIB_VERSION_LIB "5.2.40")
